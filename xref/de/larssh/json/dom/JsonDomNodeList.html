14  /**
15   * {@link List} based {@link NodeList} implementation
16   *
17   * @param <E> node element type
18   */
19  public class JsonDomNodeList<E extends Node> extends ProxiedList<E> implements NodeList {
20  	/**
21  	 * Constructor to create a {@link JsonDomNodeList} based on {@code list}.
22  	 *
23  	 * @param list list of nodes
24  	 */
25  	public JsonDomNodeList(final List<E> list) {
26  		super(unmodifiableList(list));
27  	}
28  
29  	/** {@inheritDoc} */
30  	@Override
31  	public int getLength() {
32  		return size();
33  	}
34  
35  	/** {@inheritDoc} */
36  	@NonNull
37  	@Override
38  	public E item(final int index) {
39  		return Nullables.orElseThrow(get(index));
40  	}
41  }
