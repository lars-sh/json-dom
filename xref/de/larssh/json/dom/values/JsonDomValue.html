3   import de.larssh.json.dom.JsonDomType;
4   import de.larssh.json.dom.children.JsonDomChildren;
5   
6   /**
7    * Generic wrapper to keep JSON implementation specifics off of JSON DOM.
8    *
9    * @param <T> implementation specific JSON element type
10   */
11  public interface JsonDomValue<T> {
12  	/**
13  	 * Returns a map of child node names and their values.
14  	 *
15  	 * <p>
16  	 * For JSON objects and arrays a map of keys to wrapped values should be
17  	 * returned. For other JSON elements an empty map should be returned.
18  	 *
19  	 * @return map of child node names and their value.
20  	 */
21  	JsonDomChildren<? extends JsonDomValue<T>> getChildren();
22  
23  	/**
24  	 * Returns the implementation specific JSON element.
25  	 *
26  	 * @return implementation specific JSON element
27  	 */
28  	T getJsonElement();
29  
30  	/**
31  	 * Returns the text value.
32  	 *
33  	 * <p>
34  	 * For JSON strings this is the string value without quotes. For other JSON
35  	 * elements it is their JSON representation.
36  	 *
37  	 * @return text value
38  	 */
39  	String getTextValue();
40  
41  	/**
42  	 * Returns the JSON elements value type.
43  	 *
44  	 * @return JSON data type
45  	 */
46  	JsonDomType getType();
47  }
