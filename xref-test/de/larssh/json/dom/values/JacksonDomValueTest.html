27  public class JacksonDomValueTest {
28  	/**
29  	 * Tests reading JSON and converting it to XML.
30  	 *
31  	 * @throws Exception on error
32  	 */
33  	@Test
34  	public void shouldReturnXml_whenConvertJsonToXml_givenJson() throws Exception {
35  		// given
36  		final String json = new String(
37  				Files.readAllBytes(Resources.getResourceRelativeTo(getClass(), Paths.get("data.json")).get()),
38  				StandardCharsets.UTF_8);
39  
40  		final String expectedXml = new String(
41  				Files.readAllBytes(Resources.getResourceRelativeTo(getClass(), Paths.get("data.xml")).get()),
42  				StandardCharsets.UTF_8);
43  
44  		// when
45  		final JsonNode jsonNode = new ObjectMapper().readTree(json);
46  		final Document document = new JsonDomDocument<>(new JacksonDomValue(jsonNode));
47  		final Writer writer = new StringWriter();
48  		TransformerFactory.newInstance().newTransformer().transform(new DOMSource(document), new StreamResult(writer));
49  		final String actualXml = writer.toString();
50  
51  		// then
52  		assertThat(actualXml).isEqualTo(expectedXml);
53  	}
54  }
